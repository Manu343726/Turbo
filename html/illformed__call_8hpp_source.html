39  /*
40  * The functor tml::runtime::illformed_call represents a function entity which
41  * supports ill-formed function calls to itself.
42  *
43  * Suppose we have a function f which takes two ints:
44  *
45  * void f( int , int )
46  * {
47  * ...
48  * }
49  *
50  * The call f(1,2) is well-formed (Syntactically/semantically correct), but
51  * the call f("hello") is not.
52  *
53  * 'illformed_call' supports such ill-formed calls, making them compilable.
54  * That functor wraps a function entity f, and when you call the functor:
55  *
56  * ill( "hello" );
57  *
58  * If the call is well-formed it calls the wrapped function, and if is ill-formed performs
59  * an action specified by the user. For example:
60  *
61  * auto f = tml::runtime::make_illformed_call( []( std::vector<int>& v ){ v.push_back( 1 ); } ,
62  * [](){ throw std::exception{ "Thats not a vector!!!" }; }
63  * );
64  *
65  * f();
66  *
67  * In this example, the call to f throws an exception because the call is ill-formed.
68  */
69  template<typename F , typename E>
70  struct illformed_call
71  {
72  illformed_call( F f , E else_ ) :
73  _f{ f } ,
74  _else{ else_ }
75  {}
